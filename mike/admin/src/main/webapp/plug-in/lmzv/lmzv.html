##

init:

lmzv.init({ //
formid:"form",
msgshow:"doc",
spanid:"_msg",
success:function(){alert("do something");},
error:function(){alert("do something");}
});

valid:

$obj.initV(str1,str2) //str1:"开始说明文字",str2:"验证通过显示文字" 不填或填null表示默认;
.nullV(str) //str为要显示的错误信息，不填或填null表示默认
.lenV({max:10,min:2},str)
.typeV(ts,str) //ts: 'num','user','char','email','tel','cn','en'
.comparV(o,str) //o 另一个input jquery对象$(xx)
.funV(fn,str) //fn返回true 将显示str错误信息 fn=function(k){if(k.val()...){return xx;}else{return xx;}};
.startV();

eg.:

lmzv.init({});
$("#name").initV()
.nullV()
.lenV({max:14,min:6})
.typeV("user",null)
.comparV($("#tel"))
.funV(function(k){
if(k.val()!='aaaaaa')
{return true;}
else{return false;}
},"验证失败")
.startV();
$("#tel").initV().nullV().startV();
...

lmzv.init({formid:"form1",msgshow:"alert"});

.typeV("user","格式不正确!")

.funV(function(k){
if(k.val()!='aaaaaa')
{return true;}
else{return false;}
},"验证失败")
.startV();
